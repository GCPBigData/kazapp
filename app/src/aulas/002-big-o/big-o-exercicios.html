private int buscaBinaria(int[] numeros, int inicio, int fim, int numero) {
    int indice = -1;

    while (inicio <= fim) {
        int meio = (inicio + fim) / 2;
        if (numeros[meio] < numero) {
            inicio = meio + 1;
        } else if (numeros[meio] > numero) {
            fim = meio - 1;
        } else if (numeros[meio] == numero) {
            indice = meio;
            break;
        }
    }
    return indice;
}
